برنامه مثال
ساختار فاکتوریل {
	صحیح شماره = 5 ؛
	صحیح مقدار ؛
}

ساختار ریاضی {
	ثابت اعشاری پی = ۳.۱۴ ؛
}

صحیح نتیجه ؛
منطقی الکی = غلط ؛
حرف بیمقدار ؛

صحیح محاسبه_فاکتوریل ( صحیح شماره ) {
	صحیح شمارنده = ۱ ؛
	صحیح نتیجه = ۱ ؛
	اگر شماره == ۰ یا شماره == ۱ آنگاه { برگردان ۱ ؛ }
	وگرنه {
		وقتی (شمارنده <= شماره) {
		نتیجه *= شمارنده ؛
		شمارنده++؛
		}
	}
}

صحیح اصلی () {
	فاکتوریل.مقدار = محاسبه_فانتوریل(فاکتوریل.شماره)؛

	اعشاری نمیدونم ؛

	{
		کلید ( نمیدونم )
			حالت ۱ : ؛
			پیشفرض : ؛
		تمام ؛
		بشکن ؛
	}
}
